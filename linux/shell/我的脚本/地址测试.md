```sh
#!/bin/bash

# $1 表示网段 $2表示测试组数
testip(){
    for var in {1..10};
    do
        ip=$1."$var\*$2"
        ping -c2 $ip >/dev/null 2>&1
        if [ $? = 0 ];
        then
            echo "$ip is being used" >> log
        else
            echo "$ip not be used" >> log
        fi
    done
}

for (( i=0; i<10; i++ ))
do
    {
        testip 192.168.1 $i
    } &  #将上述程序块放到后台执行
done
wait
```
